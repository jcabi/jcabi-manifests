47  /**
48   * Static reader of {@code META-INF/MANIFEST.MF} files.
49   *
50   * The class provides convenient methods to read
51   * all {@code MANIFEST.MF} files available in classpath
52   * and all attributes from them.
53   *
54   * <p>This mechanism may be very useful for transferring
55   * information from continuous integration environment to the production
56   * environment. For example, you want your site to show project version and
57   * the date of {@code WAR} file packaging. First, you configure
58   * {@code maven-war-plugin} to add this information to {@code MANIFEST.MF}:
59   *
60   * <pre> &lt;plugin>
61   *  &lt;artifactId>maven-war-plugin&lt;/artifactId>
62   *  &lt;configuration>
63   *   &lt;archive>
64   *    &lt;manifestEntries>
65   *     &lt;Foo-Version>${project.version}&lt;/Foo-Version>
66   *     &lt;Foo-Date>${maven.build.timestamp}&lt;/Foo-Date>
67   *    &lt;/manifestEntries>
68   *   &lt;/archive>
69   *  &lt;/configuration>
70   * &lt;/plugin></pre>
71   *
72   * <p>{@code maven-war-plugin} will add these attributes to your
73   * {@code MANIFEST.MF} file and the
74   * project will be deployed to the production environment. Then, you can read
75   * these attributes where it's necessary (in one of your JAXB annotated objects,
76   * for example) and show to users:
77   *
78   * <pre>  import com.jcabi.manifests.Manifest;
79   * import java.text.SimpleDateFormat;
80   * import java.util.Date;
81   * import java.util.Locale;
82   * import jakarta.xml.bind.annotation.XmlElement;
83   * import jakarta.xml.bind.annotation.XmlRootElement;
84   * &#64;XmlRootElement
85   * public final class Page {
86   *   &#64;XmlElement
87   *   public String version() {
88   *     return Manifests.read("Foo-Version");
89   *   }
90   *   &#64;XmlElement
91   *   public Date date() {
92   *     return new SimpleDateFormat("yyyy.MM.dd", Locale.ENGLISH).parse(
93   *       Manifests.read("Foo-Date");
94   *     );
95   *   }
96   * }</pre>
97   *
98   * <p>If you want to add more manifests to the collection, use
99   * its static instance:
100  *
101  * <pre>Manifests.singleton().append(new FilesMfs(new File("MANIFEST.MF")));</pre>
102  *
103  * <p>You can also modify the map directly:
104  *
105  * <pre>Manifests.singleton().put("Hello", "world");</pre>
106  *
107  * <p>The only dependency you need (check the latest version at
108  * <a href="http://manifests.jcabi.com/">jcabi-manifests</a>):
109  *
110  * <pre> &lt;dependency>
111  *  &lt;groupId>com.jcabi&lt;/groupId>
112  *  &lt;artifactId>jcabi-manifests&lt;/artifactId>
113  * &lt;/dependency></pre>
114  *
115  * @since 0.7
116  * @link <a href="http://download.oracle.com/javase/1.5.0/docs/guide/jar/jar.html#JAR%20Manifest">JAR Manifest</a>
117  * @link <a href="http://maven.apache.org/shared/maven-archiver/index.html">Maven Archiver</a>
118  * @link <a href="http://manifests.jcabi.com/index.html">manifests.jcabi.com</a>
119  * @link <a href="http://www.yegor256.com/2014/07/03/how-to-read-manifest-mf.html">How to Read MANIFEST.MF Files</a>
120  */
121 @SuppressWarnings({"PMD.TooManyMethods", "PMD.ProhibitPublicStaticMethods"})
122 public final class Manifests implements MfMap {
123 
124     /**
125      * Default singleton.
126      */
127     private static final AtomicReference<MfMap> DEFAULT =
128         new AtomicReference<>(new Manifests());
129 
130     /**
131      * Attributes retrieved.
132      */
133     private final transient Map<String, String> attributes;
134 
135     static {
136         try {
137             Manifests.singleton().append(new ClasspathMfs());
138         } catch (final IOException ex) {
139             Manifests.logLoadFailedError(ex);
140         }
141     }

143     /**
144      * Public ctor.

151     /**
152      * Public ctor.
153      * @param attrs Attributes to encapsulate
154      * @since 1.0

156     public Manifests(final Map<String, String> attrs) {
157         super();

162      * Get the singleton.
163      *
164      * @return The singleton

167     public static MfMap singleton() {
168         return Manifests.DEFAULT.get();
169     }

171     @Override
172     public int size() {
173         return this.attributes.size();
174     }
175 
176     @Override
177     public boolean isEmpty() {
178         return this.attributes.isEmpty();
179     }

181     @Override

183         return this.attributes.containsKey(key);
184     }
185 
186     @Override
187     public boolean containsValue(final String value) {
188         return this.attributes.containsValue(value);
189     }
190 
191     @Override
192     public String get(final String key) {
193         return this.attributes.get(key);
194     }
195 
196     @Override
197     public Map<String, String> getAsMap() {
198         return new HashMap<>(this.attributes);
199     }
200 
201     @Override
202     public Set<String> keySet() {
203         return new HashSet<>(this.attributes.keySet());
204     }

206     @Override
207     @SuppressWarnings("PMD.GuardLogStatement")
208     public void append(final Mfs mfs) throws IOException {
209         final long start = System.currentTimeMillis();
210         final Collection<InputStream> list = mfs.fetch();
211         int saved = 0;
212         int ignored = 0;
213         for (final InputStream stream : list) {
214             for (final Map.Entry<String, String> attr
215                 : Manifests.load(stream).entrySet()) {
216                 if (this.attributes.containsKey(attr.getKey())) {
217                     ++ignored;
218                 } else {
219                     this.attributes.put(attr.getKey(), attr.getValue());
220                     ++saved;
221                 }
222             }
223         }
224         Logger.info(
225             this,
226             // @checkstyle LineLength (1 line)
227             "%d attributes loaded from %d stream(s) in %[ms]s, %d saved, %d ignored: %[list]s",
228             this.attributes.size(), list.size(),
229             System.currentTimeMillis() - start,
230             saved, ignored,
231             new TreeSet<>(this.attributes.keySet())
232         );
233     }
234 
235     /**
236      * Read one attribute available in one of {@code MANIFEST.MF} files.
237      *
238      * <p>If such a attribute doesn't exist {@link IllegalArgumentException}
239      * will be thrown. If you're not sure whether the attribute is present or
240      * not use {@link #exists(String)} beforehand.
241      *
242      * <p>The method is thread-safe.
243      *
244      * @param name Name of the attribute
245      * @return The value of the attribute retrieved
246      */
247     public static String read(final String name) {
248         if (name == null) {
249             throw new IllegalArgumentException("attribute can't be NULL");
250         }
251         if (name.isEmpty()) {
252             throw new IllegalArgumentException("attribute can't be empty");
253         }
254         if (!Manifests.exists(name)) {
255             throw new IllegalArgumentException(
256                 Logger.format(
257                     // @checkstyle LineLength (1 line)
258                     "Attribute '%s' not found in MANIFEST.MF file(s) among %d other attribute(s): %[list]s",
259                     name,
260                     Manifests.singleton().size(),
261                     new TreeSet<>(Manifests.singleton().keySet())
262                 )
263             );
264         }
265         return Manifests.singleton().get(name);
266     }
267 
268     /**
269      * Check whether attribute exists in any of {@code MANIFEST.MF} files.
270      *
271      * <p>Use this method before {@link #read(String)} to check whether an
272      * attribute exists, in order to avoid a runtime exception.
273      *
274      * <p>The method is thread-safe.
275      *
276      * @param name Name of the attribute to check
277      * @return Returns {@code TRUE} if it exists, {@code FALSE} otherwise
278      */
279     public static boolean exists(final String name) {
280         if (name == null) {
281             throw new IllegalArgumentException("attribute name can't be NULL");
282         }
283         if (name.isEmpty()) {
284             throw new IllegalArgumentException("attribute name can't be empty");
285         }
286         return Manifests.singleton().containsKey(name);
287     }
288 
289     /**
290      * Load attributes from input stream.
291      *
292      * <p>Inside the method we catch {@code RuntimeException} (which may look
293      * suspicious) in order to protect our execution flow from expected (!)
294      * exceptions from {@link Manifest#getMainAttributes()}. For some reason,
295      * this JDK method doesn't throw checked exceptions if {@code MANIFEST.MF}
296      * file format is broken. Instead, it throws a runtime exception (an
297      * unchecked one), which we should catch in such an inconvenient way.
298      *
299      * @param stream Stream to load from
300      * @return The attributes loaded
301      * @throws IOException If some problem happens
302      * @since 0.8
303      */
304     @SuppressWarnings("PMD.AvoidCatchingGenericException")
305     private static Map<String, String> load(final InputStream stream)
306         throws IOException {
307         final ConcurrentMap<String, String> props =
308             new ConcurrentHashMap<>(0);
309         try {
310             final Manifest manifest = new Manifest(stream);
311             final Attributes attrs = manifest.getMainAttributes();
312             for (final Object key : attrs.keySet()) {
313                 final String value = attrs.getValue(
314                     Attributes.Name.class.cast(key)
315                 );
316                 props.put(key.toString(), value);
317             }
318             Logger.debug(
319                 Manifests.class,
320                 "%d attribute(s) loaded %[list]s",
321                 props.size(), new TreeSet<>(props.keySet())
322             );
323         // @checkstyle IllegalCatch (1 line)
324         } catch (final RuntimeException ex) {
325             Manifests.logLoadFailedError(ex);
326         } finally {
327             stream.close();
328         }
329         return props;
330     }
331 
332     /**
333      * Report the exception that was just thrown.
334      *
335      * @param exn The exception to report
336      */
337     private static void logLoadFailedError(final Exception exn) {
338         Logger.error(Manifests.class, "#load(): failed %[exception]s", exn);
339     }
